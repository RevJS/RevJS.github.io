Class: InMemoryBackend

The InMemoryBackend stores your model data in JavaScript objects. This backend is useful during initial development and testing but should not be used in production!

Usage example:


import {
    AutoNumberField, TextField, IntegerField,
    ModelManager, InMemoryBackend
} from 'rev-models';

class TestModel {
    @AutoNumberField({ primaryKey: true })
        item_number: number;
    @TextField()
        name: string;
    @TextField({ required: false })
        description: string;
    @IntegerField()
        score: number;

    constructor(data: Partial<TestModel>) {
        Object.assign(this, data);
    }
}

// Create a ModelManager with an InMemoryBackend
const modelManager = new ModelManager();
modelManager.registerBackend('default', new InMemoryBackend());
modelManager.register(TestModel);
